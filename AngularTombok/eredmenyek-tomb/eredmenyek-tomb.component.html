objektumfeltöltő kódja

Az ObjektumFeltolto függvény szerepe az, hogy átalakítsa a szöveges eredményeket strukturált objektumokká, amelyek könnyebben kezelhetőek és felhasználhatóak. A szöveges eredményeket általában nehezebb feldolgozni és kinyerni belőlük az információkat, mivel csak sima szövegként vannak tárolva.

ObjektumFeltolto(feltoltendoElem: string[]): versenyEredmeny[] {

  // Létrehozunk egy üres tömböt, amelyben tároljuk az eredményeket objektum formájában

  var beolvasottAdatok: versenyEredmeny[] = [];

  // Végigmegyünk a feltöltendő elemeken (szöveges eredményeken)

  for (let i: number = 0; i < feltoltendoElem.length; i++) {

    // Szétválasztjuk a szöveget szóközök mentén

    let elemDarabok: string[] = feltoltendoElem[i].split(' ')

;

    // Létrehozunk egy új objektumot a szükséges tulajdonságokkal

    let Objektum: versenyEredmeny = {

      helyezes: Number(elemDarabok[0]), // A szövegből kinyert helyezés

      meret: Number(elemDarabok[1]), // A szövegből kinyert méret

      sportag: elemDarabok[2], // A szövegből kinyert sportág

      versenyszam: elemDarabok[3], // A szövegből kinyert versenyszám

    }

    // Hozzáadjuk az objektumot az eredmények tömbhöz

    beolvasottAdatok.push(Objektum);

  }

  // Visszaadjuk az eredmények tömböt, amely már objektumokat tartalmaz

  return beolvasottAdatok;

}
